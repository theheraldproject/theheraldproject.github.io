131  template <typename T>
132  void addEntropy(T entropy) {
133  // Get size of amount of entropy we have
134  constexpr std::size_t size = sizeof(T);
135  // TODO consider whether there's benefit to detecting most significant set bits.
136  // This may provide a benefit if the method of combination isn't XOR.
137 
138  // See if we are at multiples of std::size_t
139  constexpr std::size_t multiple = sizeof(std::size_t) / size; // integer division, rounds down
140  // Note the above cannot ever be 0 because std::size_t is always the largest size on a given architecture
141 
142  std::size_t toXor = 0;
143  for (std::size_t i = 0;i < multiple;i++) {
144  if (0 != i) {
145  toXor << size;
146  }
147  toXor += entropy;
148  }
149 
150  // XOR will ensure the same distribution of 0 and 1 over multiple applications
151  m_entropy = m_entropy ^ toXor;
152  }
153 
154 
155  std::string methodName() const {
156  return m_source->methodName();
157  }
158 
159  void nextBytes(std::size_t count, Data& into) {
160  constexpr std::size_t byteSize = sizeof(std::byte);
161  constexpr std::size_t sizeTSize = sizeof(std::size_t);
162  constexpr std::size_t shifts = sizeTSize / byteSize;
163  Data sourcedInto;
164  m_source->nextBytes(count,sourcedInto);
165 
166  std::vector<std::byte> result;
167 
168  // now add in entropy
169  for (std::size_t byteIndex = 0;byteIndex < count;byteIndex++) {
170  result.push_back((std::byte)(
171  std::size_t(sourcedInto.at(byteIndex))
172  ^
173  (m_entropy >> 8 * (byteIndex % shifts))
174  ));
175  }
176 
177  Data final(result);
178  into.append(final);
179  }
180 
181  int nextInt() {
182  return (int)(m_source->nextInt() ^ m_entropy);
183  }
184 
185  double nextDouble() {
186  return (double)(((std::size_t)m_source->nextDouble()) ^ m_entropy);
187  }
188 
189 private:
190  std::unique_ptr<RandomnessSource> m_source;
191  std::size_t m_entropy;
192 };
